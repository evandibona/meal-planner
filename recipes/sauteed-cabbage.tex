\section{ Sauteed Cabbage }

\subsection{ Ingredients }

\begin{itemize}
  \item Salt
  \item Pepper
  \item Garlic Powder
  \item 3 ounces of Green Cabbage
  \item A bit of Butter
\end{itemize}

\subsection{ Directions }

\begin{enumerate}
  \item Slice thin, sautee cabbage in butter, season to taste. 
\end{enumerate}
